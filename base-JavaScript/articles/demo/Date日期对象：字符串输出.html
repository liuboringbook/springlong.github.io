Date日期对象：字符串输出

执行代码：

var dt = new Date("2014/8/10 8:00:00");
document.write("<p><b>.toString()</b></br>" + dt.toString());
document.write("<p><b>.toDateString()</b></br>" + dt.toDateString());
document.write("<p><b>.toTimeString()</b></br>" + dt.toTimeString());
document.write("<p><b>.toLocaleString()</b></br>" + dt.toLocaleString());
document.write("<p><b>.toLocaleDateString()</b></br>" + dt.toLocaleDateString());
document.write("<p><b>.toLocaleTimeString()</b></br>" + dt.toLocaleTimeString());
document.write("<p><b>.toUTCString()</b></br>" + dt.toUTCString());
document.write("<p><b>.toGMTString()</b></br>" + dt.toGMTString());
