Date日期对象：日期对象的创建

执行代码：

var dt1 = new Date("2014/8/10 8:00:00");
var dt2 = new Date("2014/9/10 8:00:00");

//获取Date对象的毫秒级快照
document.write("<p>dt1的毫秒级快照是：" + +dt1);
document.write("<p>dt2的毫秒级快照是：" + +dt2);

//计算两个Date对象之间相差的毫秒级快照
document.write("<p>dt1和dt2之间相差的毫秒数为：" + (dt2 - dt1));

//判断两个Date对象的大小关系
if(dt2 > dt1){
    document.write("<p>dt2的日期时间位于dt1之后！");
}else{
    document.write("<p>dt2的日期时间位于dt1之前！");
}
