日期操作：判断一个年份是否为闰年

运行测试

执行代码

document.getElementById("btnCheck").onclick = function()
{
    var frmYear = document.getElementById("inputYear"),
        year = parseInt(frmYear.value);

    //判断正确的年份值
    if(isNaN(year)){
        alert("你输入的年份有误，请重新输入！");
        frmYear.focus();
        frmYear.select();
        return;
    }
    frmYear.value = year;  //去掉文本框中的前缀“0”

    //判断是否为闰年（普通年能被4整除且不能被100整除的年份是闰年，世纪年能被400整除的年份是闰年）
    if(year % 400 === 0 || year % 4 === 0 && year % 100 !== 0){
        alert("判断结果：闰年！");
    }else{
        alert("判断结果：不是闰年！");
    }
};
